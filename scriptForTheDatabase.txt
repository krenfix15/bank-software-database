CREATE SEQUENCE seq_clienti_PM1
START WITH 100000
INCREMENT BY 1;

CREATE SEQUENCE seq_conturi_PM1
START WITH 100000
INCREMENT BY 1;

CREATE SEQUENCE seq_locatii_PM1
START WITH 100
INCREMENT BY 10;

CREATE SEQUENCE seq_carduri_PM1
START WITH 100
INCREMENT BY 100;

CREATE TABLE carduri_PM1 (
ID_CARD INTEGER NOT NULL PRIMARY KEY,
Tip VARCHAR2(20) NOT NULL UNIQUE
);

CREATE TABLE locatii_PM1 (
ID_LOCATIE INTEGER NOT NULL PRIMARY KEY,
Judet VARCHAR2(15) NOT NULL UNIQUE,
Abreviere VARCHAR2(2) NOT NULL UNIQUE
);

CREATE TABLE clienti_PM1 (
ID_CLIENT INTEGER NOT NULL,
Nume VARCHAR2(30) NOT NULL,
Prenume VARCHAR2(30) NOT NULL,
CNP VARCHAR2(13) NOT NULL UNIQUE,
DataNastere DATE NOT NULL,
Email VARCHAR2(40),
Telefon VARCHAR2(10),
Judet VARCHAR2(15) CONSTRAINT fk_clienti_judet REFERENCES locatii_PM1(Judet),
PRIMARY KEY(ID_CLIENT, CNP)
);

CREATE TABLE conturi_PM1 (
ID_CONT INTEGER NOT NULL PRIMARY KEY,
CNP VARCHAR2(13) CONSTRAINT fk_conturi_cnp REFERENCES clienti_PM1(CNP),
Perioada INTEGER NOT NULL,
Sold INTEGER NOT NULL,
DataDepunere DATE NOT NULL,
Dobanda NUMBER(5) NOT NULL,
TipCard VARCHAR2(20) CONSTRAINT fk_conturi_tipcard REFERENCES carduri_PM1(Tip)
);
